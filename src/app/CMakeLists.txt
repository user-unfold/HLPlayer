find_package(Qt6 REQUIRED COMPONENTS Core Gui Qml Quick QuickControls2)

add_executable(hlplayer_app main.cpp
    ${CMAKE_SOURCE_DIR}/resources/resources.qrc
)

target_compile_features(hlplayer_app PRIVATE cxx_std_20)

set_target_properties(hlplayer_app PROPERTIES
    AUTOMOC ON
    AUTORCC ON
)

target_link_libraries(hlplayer_app PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Qml
    Qt6::Quick
    Qt6::QuickControls2
    hlplayer_core
    hlplayer_qml
)

# Standard warnings
if(MSVC)
    target_compile_options(hlplayer_app PRIVATE /W4)
else()
    target_compile_options(hlplayer_app PRIVATE -Wall -Wextra)
endif()

# Post-build: copy sibling library DLLs to app directory so the executable
# can find them at runtime.  Uses cmake -E copy_if_different to avoid
# unnecessary file touches.
set(_HLPLAYER_DLLS
    $<TARGET_FILE:hlplayer_core>
    $<TARGET_FILE:hlplayer_extractor>
    $<TARGET_FILE:hlplayer_render>
    $<TARGET_FILE:hlplayer_qml>
)

add_custom_command(TARGET hlplayer_app POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E make_directory "$<TARGET_FILE_DIR:hlplayer_app>/qml/HLPlayer"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different ${_HLPLAYER_DLLS} "$<TARGET_FILE_DIR:hlplayer_app>"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different "$<TARGET_FILE:hlplayer_qml>" "$<TARGET_FILE_DIR:hlplayer_app>/qml/HLPlayer"
    # Sync QML module metadata (qmldir, qmltypes) generated by qt6_add_qml_module
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "$<TARGET_PROPERTY:hlplayer_qml,BINARY_DIR>/HLPlayer/qmldir"
        "$<TARGET_FILE_DIR:hlplayer_app>/qml/HLPlayer/qmldir"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "$<TARGET_PROPERTY:hlplayer_qml,BINARY_DIR>/HLPlayer/hlplayer_qml.qmltypes"
        "$<TARGET_FILE_DIR:hlplayer_app>/qml/HLPlayer/hlplayer_qml.qmltypes"
    # Deploy SDL2 runtime dependency
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CURRENT_SOURCE_DIR}/../../SDL2-2.30.8/x86_64-w64-mingw32/bin/SDL2.dll"
        "$<TARGET_FILE_DIR:hlplayer_app>"
    # Deploy MinGW UCRT64 runtime DLLs (libstdc++, libgcc, libwinpthread, libgomp for OpenMP)
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CXX_COMPILER}/../libstdc++-6.dll"
        "$<TARGET_FILE_DIR:hlplayer_app>"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CXX_COMPILER}/../libgcc_s_seh-1.dll"
        "$<TARGET_FILE_DIR:hlplayer_app>"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CXX_COMPILER}/../libwinpthread-1.dll"
        "$<TARGET_FILE_DIR:hlplayer_app>"
    COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${CMAKE_CXX_COMPILER}/../libgomp-1.dll"
        "$<TARGET_FILE_DIR:hlplayer_app>"
    COMMENT "Deploying HLPlayer libraries to app directory"
)
